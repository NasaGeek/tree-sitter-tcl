# Valid (but I wouldn't say idiomatic) Tcl that we can't parse.
expr 1 + 1
expr 1 {+} 1
expr {1+1}
set(a)(b) 1
# Tcl treats the last ) as a literal char
puts $arr())
# Tcl treats the first ) as a literal char
set arr()) 0
# Tcl treats $ at the end as a lteral char
set $arr($)
set arr($) 0
# Empty array name is valid
set (a) 0
puts $(a)
